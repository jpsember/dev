_TOSEMI: $_NONSEMI ;

_OPENEX: \{ \~
_CLOSEEX: \~ \}
_ANY: .*
_NOLF: [^\n]*
_OPTLF: \n?

   ( $_ANY -- (  $_ANY ( \n  | $_OPENEX ) $_ANY ) ) \
        $_OPENEX   \
        $_NOLF $_OPTLF \
        (($_ANY -- (  $_ANY ( \n  | $_CLOSEEX ) $_ANY ) ) \n)* \
        (( $_ANY -- (  $_ANY ( \n  | $_CLOSEEX ) $_ANY ) ) $_CLOSEEX $_NOLF $_OPTLF)
